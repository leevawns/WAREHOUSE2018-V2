CREATE SCHEMA `warehouse_sourcedata` ;


CREATE TABLE `warehouse_sourcedata`.`item_information` (
  `PRODUCT` VARCHAR(45) NOT NULL COMMENT '',
  `ITEM_CODE` VARCHAR(45) NOT NULL COMMENT '',
  `PART_NAME` VARCHAR(100) NOT NULL COMMENT '',
  `MAKER` VARCHAR(100) NOT NULL COMMENT '',
  `PART_NO` VARCHAR(100) NOT NULL COMMENT '',
  `LOT_SIZE` DOUBLE NOT NULL COMMENT '',
  `UNIT` VARCHAR(20) NULL COMMENT '',
  `LENGTH-OFFSET` VARCHAR(20) NULL COMMENT '',
  `BARCODE` VARCHAR(100) NULL COMMENT '',
  `PROCESS` VARCHAR(100) NULL COMMENT '',
  `INVENTORY_LOCAL` VARCHAR(100) NULL COMMENT '',
  `TARE` DOUBLE NULL COMMENT '',
  `SAMPLE` DOUBLE NULL COMMENT '',
  PRIMARY KEY (`ITEM_CODE`)  COMMENT '')





CREATE TABLE `warehouse_sourcedata`.`invoice_information` (
  `INVOICE_NO` VARCHAR(200) NOT NULL COMMENT '',
  `INVOICE_DATE` DATETIME NOT NULL COMMENT '',
  `ECODE` VARCHAR(45) NOT NULL COMMENT '',
  PRIMARY KEY ( `INVOICE_NO`)  COMMENT '');


CREATE TABLE `warehouse_sourcedata`.`invoice_no` (
`NO` INT(20) NOT NULL AUTO_INCREMENT COMMENT '' ,
  `ITEM_CODE` VARCHAR(50) NOT NULL COMMENT '',
  `ITEM` VARCHAR(100) NULL COMMENT '',
  `MAKER` VARCHAR(100) NULL COMMENT '',
  `MAKER_CODE` VARCHAR(100) NULL COMMENT '',
  `LOT_SIZE` DOUBLE NULL COMMENT '',
  `UNIT` VARCHAR(20) NULL COMMENT '',
  `QUANTITY` DOUBLE NULL COMMENT '',
  `INPUT` DOUBLE NULL COMMENT '',
  `BALANCE` DOUBLE NULL COMMENT '',
  `INVOICE_NO` VARCHAR(100) NOT NULL COMMENT '',
  PRIMARY KEY (`NO`, `ITEM_CODE`, `INVOICE_NO`)  COMMENT '');

  `QUANTITY` DOUBLE NULL COMMENT '',
  `INVOICE_ECODE` VARCHAR(20) NULL COMMENT '',
  `INPUT_NO` VARCHAR(100) NULL COMMENT '',
  `INPUT_DATE` DATETIME NULL COMMENT '',
  `INPUT_ECODE` VARCHAR(20) NULL COMMENT '',
  `LINEBACK_NO` VARCHAR(100) NULL COMMENT '',
  `LINEBACK_DATE` DATETIME NULL COMMENT '',
  `LINEBACK_ECODE` VARCHAR(20) NULL COMMENT '',
  `OUTPUT_NO` VARCHAR(100) NULL COMMENT '',
  `OUTPUT_DATE` DATETIME NULL COMMENT '',
  `OUTPUT_ECODE` VARCHAR(20) NULL COMMENT '',
  `DESTINATION` VARCHAR(100) NULL COMMENT '',);

  `NO` INT NOT NULL AUTO_INCREMENT COMMENT '',

  `PART_NAME` VARCHAR(100) NULL COMMENT '',
  `MAKER` VARCHAR(100) NULL COMMENT '',
  `PART_NO` VARCHAR(100) NULL COMMENT '',
  `LOT_SIZE` DOUBLE NULL COMMENT '',

  `STOCK` DOUBLE NULL COMMENT '',
